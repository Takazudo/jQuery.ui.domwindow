jQuery.ui.domwindow - demo - widgetlike

widgetlike

Dialogs can be handled like jQuery.ui.dialog

$(function(){
    var $dialog = $('#iddialogcontent').domwindow({
        width: 700,
        height: 600,
        open: function(event, data){
            console.log('opened1!', data.dialog);
        },
        close: function(){
            console.log('closed1!');
        }
    });
    $dialog.bind('domwindow.open', function(e, data){
        console.log('opened2!', data.dialog);
    });
    $dialog.bind('domwindow.close', function(){
        console.log('closed2!');
    });
    $('#widgetlike-open').click(function(){
        $dialog.domwindow('open');
    });
    $(document).on('click', '.widgetlike-close', function(){
        $dialog.domwindow('close');
    });
});

jQuery.ui.domwindow - demo - api

ajaxdialog

$(function(){
    $('#apiex-ajax-open').click(function(){
        window.domwindowApi.open('dialog1.html', {
            ajaxdialog: true,
            width: 400,
            height: 600,
            callback: function($dialog){
                console.log('opend!', $dialog);
            }
        });
    });
});
